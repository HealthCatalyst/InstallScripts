#!/bin/sh
set -e

echo "Version 1.1"

#
# This script is meant for quick & easy install via:
#   'curl -sSL https://healthcatalyst.github.io/InstallScripts/nlp/installsolr.txt | sh'

# delete old versions
echo "stopping existing docker container"
docker stop fabric.nlp.docker.solr || echo 'no container to stop'
echo "removing docker container"
docker rm fabric.nlp.docker.solr || echo 'no container to remove'
echo "removing docker image"
docker rmi healthcatalyst/fabric.nlp.docker.solr || echo 'no image to remove'


docker run -p 3306:3306 -d --name fabric.nlp.docker.solr -t healthcatalyst/fabric.nlp.docker.solr

echo "sleeping until docker container is up"
until [ "`/usr/bin/docker inspect -f {{.State.Running}} fabric.nlp.docker.solr`"=="true" ]; do
    sleep 1s;
done;
