#!/bin/bash
# curl -sSL https://imranq2.github.io/InstallScripts/setupdrives.txt | sh

echo "Version 1.1"
set -x

echo "creating disk 1"
sudo fdisk /dev/sdc <<EEOF
d
n
p
1


w
EEOF


sudo mkfs -t ext4 /dev/sdc1
sudo mkdir /mnt/data1
sudo mount /dev/sdc1 /mnt/data1

mkdir -p /mnt/data1/db
mkdir -p /mnt/data1/log

sudo chown -R elasticsearch:elasticsearch /mnt/data1/

sudo chown -R elasticsearch:elasticsearch /mnt/data1/log
sudo chown -R elasticsearch:elasticsearch /mnt/data1/db

echo "Creating disk 2"
sudo fdisk /dev/sdd <<EEOF
d
n
p
1


w
EEOF


sudo mkfs -t ext4 /dev/sdd1
sudo mkdir /mnt/data2
sudo mount /dev/sdd1 /mnt/data2

mkdir -p /mnt/data2/db
mkdir -p /mnt/data2/log

sudo chown -R elasticsearch:elasticsearch /mnt/data2/

sudo chown -R elasticsearch:elasticsearch /mnt/data2/log
sudo chown -R elasticsearch:elasticsearch /mnt/data2/db

lsblk

set +x
