#!/bin/sh
set -e
#
# This script is meant for quick & easy install via:
#   'curl -sSL https://imranq2.github.io/InstallScripts/testsql.txt | sh -s <server>'

servername="$1"

sqlcmd -S $servername -Q 'SET NOCOUNT ON; SELECT SUSER_NAME()' -h-1

sqlcmd -S $servername -Q 'SET NOCOUNT ON;select * from sys.databases' -h-1
