#!/bin/bash
# curl -sSL https://imranq2.github.io/InstallScripts/setupdrives.txt | sh

echo "Version 1.0"

echo "creating disk 1"
sudo fdisk /dev/sdc <<EEOF
d
n
p
1


w
EEOF
exit 0

sudo mkfs -t ext4 /dev/sdc1
sudo mkdir /mnt/data1
sudo mount /dev/sdc1 /mnt/data1

echo "Creating disk 2"
sudo fdisk /dev/sdd <<EEOF
d
n
p
1


w
EEOF
exit 0

sudo mkfs -t ext4 /dev/sdd2
sudo mkdir /mnt/data2
sudo mount /dev/sdd2 /mnt/data2
