#!/bin/sh
set -e

echo "Version 1.2"

#
# This script is meant for quick & easy install via:
#   'curl -sSL https://healthcatalyst.github.io/InstallScripts/nlp/installweb.txt | sh -s <mysqlserver> <solrserver> <jobserver>'

# Remember: no spaces allowed in variable set commands in bash
mysqlserver="$1"
solrserver="$2"
jobserver="$3"

echo "mysqlserver:"$mysqlserver
echo "solrserver:"$solrserver
echo "jobserver:"$jobserver

# delete old versions
echo "stopping existing docker container"
docker stop fabric.nlp.docker.web || echo 'no container to stop'
echo "removing docker container"
docker rm fabric.nlp.docker.web || echo 'no container to remove'
echo "removing docker image"
docker rmi healthcatalyst/fabric.nlp.docker.web || echo 'no image to remove'


docker run -p 8083:8083 -d --name fabric.nlp.docker.web --add-host mysqlserver:$mysqlserver --add-host solrserver:$solrserver --add-host jobserver:$jobserver -t healthcatalyst/fabric.nlp.docker.web

echo "sleeping until docker container is up"
until [ "`/usr/bin/docker inspect -f {{.State.Running}} fabric.nlp.docker.web`"=="true" ]; do
    sleep 1s;
done;
