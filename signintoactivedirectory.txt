#!/bin/sh
set -e
#
# This script is meant for quick & easy install via:
#   'curl -sSL https://imranq2.github.io/InstallScripts/signintoactivedirectory.txt | sh -s <username> <domain>'

username="$1"
domain="$2"
uppercasedomain=${domain^^}
usernameplusdomain="$username@$uppercasedomain"

kinit -kt user.keytab $usernameplusdomain -V
